import requests

API_KEY = "your_api_key"
API_URL = "https://api.jeniya.com/v1/chat/completions"  # 简易API中转地址

headers = {
    "Authorization": f"Bearer {API_KEY}",
    "Content-Type": "application/json"
}

data = {
    "model": "gpt-4o",  # 使用高级模型
    "messages": [
        {"role": "system", "content": "你是一个专业助手"},
        {"role": "user", "content": "请介绍简易API中转站的优势"}
    ],
    "temperature": 0.7,
    "max_tokens": 1500
}

response = requests.post(API_URL, headers=headers, json=data)
print(response.json())
